import {OpaqueToken} from "@angular/core";
export const OtherService = new OpaqueToken("OtherService");

export function otherServiceFactory(i: any) {
  return i.get('otherService');
}
export const otherServiceProvider = {
  provide: OtherService,
  useFactory: otherServiceFactory,
  deps: ['$injector']
};

export class MyService {
  constructor(@Inject(OtherService) private other) {}
}
